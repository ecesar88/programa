Query
query GetClientById($id: Int!) {
  getClientById(id: $id) {
    ... on BaseError {
      message
    }
    ... on QueryGetClientByIdSuccess {
      data {
        ...ClientFragment
      }
    }
    ... on RecordNotFoundError {
      message
    }
  }
}
